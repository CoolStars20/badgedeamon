% Macro BADGE
% This macro sets one side of one badge.
% It gets called twice below (once for the front side and once of the back side).
% Modify text or dimensions here to make you badge fit your format.

{% macro badge(image, title, dat) -%}
\noindent
\begin{minipage}[h][6.2in][c]{4.2in}
{{title}}\\
\vspace{.2in}
\noindent
\fbox{
\begin{minipage}[h][6.2in][c]{4.2in}
\noindent
\begin{minipage}[h][.78in][b]{4.2in}
\begin{centering}
\includegraphics[height=.78in, width=\textwidth,
          keepaspectratio]{csheader}\\
\end{centering}
\end{minipage}
{% if dat['role'] %}
\begin{minipage}[t][4.6in][c]{4.2in}
{% else %}
\begin{minipage}[t][5.2in][c]{4.2in}
{% endif %}
\begin{centering}
\vspace{ \stretch{.3} }
{% if dat['pronoun'] %}
\large {{dat['pronoun']}}
\vspace{ \stretch{.1} }
{% endif %}
\fontsize{1cm}{1.cm}\selectfont
\textbf{ {{dat['name']}} }\\[2mm]
\vspace{ \stretch{.1} }
\Large \textbf{ {{dat['inst'] | default('affiliation unkown')}} }\\
\vfill
{% if dat['typetext'] %}
\includegraphics[height=3.2in, width=0.92\textwidth,
 keepaspectratio]{{'{'}}{{dat[image]}}{{'}'}}\\
{% else %}
         \includegraphics[height=3.8in, width=0.92\textwidth,
          keepaspectratio]{{'{'}}{{dat[image]}}{{'}'}}\\
 {% endif %}
        \vspace{ \stretch{.5} }
      \end{centering}
    \end{minipage}
    {% if dat['typetext'] %}
     \begin{minipage}[t][0.6in][c]{4.2in}
       \begin{tcolorbox}[width=0.98\textwidth,colback={{'{'}}{{dat['rolecolor']}}{{'}'}},colupper=white]
           \centering
           {\bf \huge {{dat['role']}} }
         \end{tcolorbox}
       \end{minipage}
        {% endif %}
  \end{minipage}
}\end{minipage}%
{%- endmacro %}
